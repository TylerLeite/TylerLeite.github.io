<link rel="import" href="/bower_components/polymer/polymer-element.html">
<link rel="import" href="/bower_components/iron-pages/iron-pages.html">

<dom-module id="my-src">
  <template>
    <style>
      #nav {
        display: flex;
        flex-flow: row nowrap;
        align-items: center;
        justify-content: space-around;

        width: 20vw;
      }

      #nav > p {
        height: 100%;
        cursor: pointer;
      }
      #nav > p:hover {
        color: #888;
      }

      /* Define the animation here. You can do pretty much anything you want. */
      @keyframes slide {
        from {
          left: -70%;
        }

        to {
          left: 0;
        }
      }

      /* Adding this class starts the animation, removing it cuts it short */
      .animated {
        animation-duration: 1s;
        animation-name: slide;
      }

      .page {
        position: relative;

        width: 100vw;
        height: 100vh;

        display: flex;
        align-items: center;
        justify-content: center;
      }

      .page > p {
        width: 30vw;
        height: 15vh;

        display: flex;
        align-items: center;
        justify-content: center;

        border: 3px solid black;

        font-size: 5vmin;
      }

      .page.one > p {
        border-color: #dd4444;
      }
      .page.two > p {
        border-color: #44dd44;
      }
      .page.three > p  {
        border-color: #4444dd;
      }
    </style>

    <div id="nav">
      <!-- In case you were wondering how to implement a nav bad, I find this to be the simplest way -->
      <p onclick="select(0);">Page 1</p>
      <p onclick="select(1);">Page 2</p>
      <p onclick="select(2);">Page 3</p>
    </div>

    <iron-pages
      id="pages"
      selected="{{selected}}"
    >
      <div class="page one"><p>My page 1</p></div>
      <div class="page two"><p>My page 2</p></div>
      <div class="page three"><p>My page 3</p></div>
    </iron-pages>
  </template>

  <script>
    function select (n) {
      /* Be careful with the shadow-dom, 2.0 doesn't mess around */
      /* Depending on how deep this element is, you may need to find a
         different way of accessing iron-pages */
      let pages = document.querySelector('my-src').$.pages;
      pages.select(n);
    }

    // Replace mySrc and my-src with appropriate names if copy-pasting this code
    class mySrc extends Polymer.Element {
      static get is () { return 'my-src'; }

      static get properties () {
        return {
          // Two-way bound to iron-pages selected property
          selected: {
            type: Number,
            notify: true,
            value: 0,
            observer: '_newPageSelected',
          },
        };
      }

      // Observers
      _newPageSelected (newVal, oldVal) {
        let pages = this.$.pages.children;

        // Run the animation on the newly selected page
        if (!pages[newVal].className.includes('animated')) {
          pages[newVal].className += ' animated';
        }

        if (typeof oldVal !== 'undefined') {
          // Stop the animation of hidden pages
          pages[oldVal].className = pages[oldVal].className.split(' animated').join('');
        }
      }
    }

    window.customElements.define(mySrc.is, mySrc);
  </script>
</dom-module>
